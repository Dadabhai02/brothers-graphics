// ===== DATABASE (IndexedDB) =====
const DB_NAME = 'PrintCraftDB';
const DB_VERSION = 1;
let db;

function initDB() {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onupgradeneeded = (e) => {
      const d = e.target.result;
      if (!d.objectStoreNames.contains('cart')) {
        d.createObjectStore('cart', { keyPath: 'id' });
      }
      if (!d.objectStoreNames.contains('liked')) {
        d.createObjectStore('liked', { keyPath: 'id' });
      }
      if (!d.objectStoreNames.contains('products')) {
        const ps = d.createObjectStore('products', { keyPath: 'id', autoIncrement: true });
        ps.createIndex('category', 'category', { unique: false });
      }
    };
    req.onsuccess = (e) => { db = e.target.result; resolve(db); };
    req.onerror = reject;
  });
}

function dbGet(store, key) {
  return new Promise((resolve) => {
    const tx = db.transaction(store, 'readonly');
    const req = tx.objectStore(store).get(key);
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => resolve(null);
  });
}
function dbPut(store, val) {
  return new Promise((resolve) => {
    const tx = db.transaction(store, 'readwrite');
    tx.objectStore(store).put(val);
    tx.oncomplete = resolve;
  });
}
function dbDelete(store, key) {
  return new Promise((resolve) => {
    const tx = db.transaction(store, 'readwrite');
    tx.objectStore(store).delete(key);
    tx.oncomplete = resolve;
  });
}
function dbGetAll(store) {
  return new Promise((resolve) => {
    const tx = db.transaction(store, 'readonly');
    const req = tx.objectStore(store).getAll();
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => resolve([]);
  });
}

// ===== CART =====
async function addToCart(id) {
  const product = products.find(p => p.id === id);
  if (!product) return;
  const existing = await dbGet('cart', id);
  if (existing) {
    existing.qty += 1;
    await dbPut('cart', existing);
  } else {
    await dbPut('cart', { id, name: product.name, emoji: product.emoji, price: product.price, qty: 1 });
  }
  cart[id] = existing ? existing : { id, name: product.name, emoji: product.emoji, price: product.price, qty: 1 };
  if (existing) cart[id].qty = existing.qty;
  await syncCart();
  showToast(`🛒 ${product.name} added to cart!`);
}

async function syncCart() {
  const items = await dbGetAll('cart');
  cart = {};
  items.forEach(i => cart[i.id] = i);
  updateCartUI();
}

function updateCartUI() {
  const items = Object.values(cart);
  const total = items.reduce((s, i) => s + i.price * i.qty, 0);
  const count = items.reduce((s, i) => s + i.qty, 0);
  document.getElementById('cartCount').textContent = count;

  if (items.length === 0) {
    document.getElementById('cartItems').innerHTML = `
      <div class="empty-cart">
        <div class="empty-cart-icon">🛒</div>
        <p>Your cart is empty</p>
        <p style="font-size:0.8rem;margin-top:0.5rem">Add some beautiful products!</p>
      </div>`;
    document.getElementById('cartFooter').innerHTML = '';
    return;
  }

  document.getElementById('cartItems').innerHTML = items.map(i => `
    <div class="cart-item">
      <div class="cart-item-img">${i.emoji}</div>
      <div class="cart-item-info">
        <div class="cart-item-name">${i.name}</div>
        <div class="cart-item-price">₹${i.price}/- × ${i.qty} = ₹${i.price * i.qty}/-</div>
        <div class="cart-item-qty">
          <button class="qty-btn" onclick="changeQty(${i.id}, -1)">−</button>
          <span class="qty-num">${i.qty}</span>
          <button class="qty-btn" onclick="changeQty(${i.id}, 1)">+</button>
        </div>
      </div>
      <button class="remove-item" onclick="removeFromCart(${i.id})">🗑</button>
    </div>
  `).join('');

  document.getElementById('cartFooter').innerHTML = `
    <div class="cart-total">
      <span class="cart-total-label">Total (${count} items)</span>
      <span class="cart-total-value">₹${total}/-</span>
    </div>
    <button class="checkout-btn" onclick="showToast('🎉 Order placed! We will contact you shortly.')">Proceed to Checkout</button>
  `;
}

async function changeQty(id, delta) {
  const item = cart[id];
  if (!item) return;
  item.qty += delta;
  if (item.qty <= 0) {
    await removeFromCart(id);
  } else {
    await dbPut('cart', item);
    await syncCart();
  }
}

async function removeFromCart(id) {
  await dbDelete('cart', id);
  delete cart[id];
  await syncCart();
}

function toggleCart() {
  const overlay = document.getElementById('cartOverlay');
  const sidebar = document.getElementById('cartSidebar');
  overlay.classList.toggle('open');
  sidebar.classList.toggle('open');
}

// ===== TOAST =====
let toastTimer;
function showToast(msg) {
  const t = document.getElementById('toast');
  document.getElementById('toastMsg').textContent = msg;
  t.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => t.classList.remove('show'), 3000);
}

function scrollTo(hash) {
  document.querySelector(hash).scrollIntoView({ behavior: 'smooth' });
}

  // Load cart
  await syncCart();
  // Render everything
  renderCategories();
  renderFilterBar();
  renderProducts();
  renderGallery();
  renderTestimonials();
}
